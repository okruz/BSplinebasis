139 template <typename T, size_t sizea, size_t sizeb>
140 static std::array<T, std::max(sizea, sizeb)> &add(std::array<T, sizea> &a,
141 std::array<T, sizeb> &b) {
142 if constexpr (sizeb > sizea) {
143 return add(b, a);
144 } else {
145 for (size_t i = 0; i < sizeb; i++) {
146 a[i] += b[i];
147 }
148 return a;
149 }
150 }

159 OperatorSum(O1 o1, O2 o2) : _o1(std::move(o1)), _o2(std::move(o2)){};

167 static constexpr size_t outputOrder(size_t inputOrder) {
168 return std::max(O1::outputOrder(inputOrder), O2::outputOrder(inputOrder));

184 template <typename T, size_t size>
185 std::array<T, outputOrder(size - 1) + 1> transform(
186 const std::array<T, size> &input, const support::Grid<T> &grid,
187 size_t intervalIndex) const {
188 auto a = _o1.transform(input, grid, intervalIndex);
189 auto b = _o2.transform(input, grid, intervalIndex);
190
191 // Negate b if subtraction is requested.
192 if constexpr (operation == AdditionOperation::SUBTRACTION) {
193 for (T &el : b) {
194 el *= static_cast<T>(-1);
195 }
196 }
197
198 return add(a, b);
199 }
