39 template <typename T, size_t size>
40 static T evaluateInterval(const std::array<T, size> &a, const T &dxhalf) {
41 // Use Horner's scheme to evaluate.
42 constexpr int endIndex =
43 static_cast<int>(size) - static_cast<int>(size % 2 == 0 ? 2 : 1);
44 static_assert(endIndex >= 0);
45
46 T result = a.at(endIndex) / static_cast<T>(endIndex + 1);
47 const T dxhalf_squared = dxhalf * dxhalf;
48
49 for (int i = endIndex - 2; i >= 0; i -= 2) {
50 result = dxhalf_squared * result + a[i] / static_cast<T>(i + 1);
51 }
52 return static_cast<T>(2) * dxhalf * result;
53 }

76 template <typename T, size_t order>
77 T evaluate(const Spline<T, order> &a) const {
78 const size_t nintervals = a.getSupport().numberOfIntervals();
79
80 T result = static_cast<T>(0);
81
82 for (size_t i = 0; i < nintervals; i++) {
83 const size_t absIndex = a.getSupport().absoluteFromRelative(i);
84 const T dxhalf =
85 (a.getSupport()[i + 1] - a.getSupport()[i]) / static_cast<T>(2);
86
87 result +=
88 evaluateInterval(_o.transform(a.getCoefficients()[i],
89 a.getSupport().getGrid(), absIndex),
90 dxhalf);
91 }
92 return result;
93 }

99 template <typename T, size_t order>
100 T operator()(const Spline<T, order> &a) const {
101 return evaluate(a);
102 }
