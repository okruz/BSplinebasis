70 void checkValidity() const {
71 const bool isEmpty = _startIndex == 0 && _endIndex == 0;
72 const bool containsElement = _endIndex > _startIndex;
73 const bool withinBounds = _endIndex <= _grid.size();
74
75 const bool valid = isEmpty || (containsElement && withinBounds);
76
77 if (!valid) {
78 throw BSplineException(ErrorCode::INCONSISTENT_DATA);
79 }
80 }

432 Support calcUnion(const Support &s) const {
433 DURING_TEST_CHECK_VALIDITY();
434 DURING_TEST_CHECK_VALIDITY_OF(s);
435 if (!hasSameGrid(s)) {
436 throw BSplineException(ErrorCode::DIFFERING_GRIDS);
437 }
438
439 const bool thisEmpty = empty();
440 const bool sEmpty = s.empty();
441
442 if (thisEmpty && sEmpty) {
443 // Both Supports are empty, return empty Support.
444 return createEmpty(_grid);
445 } else if (thisEmpty && !sEmpty) {
446 return s;
447 } else if (!thisEmpty && sEmpty) {
448 return *this;
449 }
450
451 const size_t newStartIndex = std::min(_startIndex, s._startIndex);
452 const size_t newEndIndex = std::max(_endIndex, s._endIndex);
453 return Support(_grid, newStartIndex, newEndIndex);
454 };

465 Support calcIntersection(const Support &s) const {
466 DURING_TEST_CHECK_VALIDITY();
467 DURING_TEST_CHECK_VALIDITY_OF(s);
468 if (!hasSameGrid(s)) {
469 throw BSplineException(ErrorCode::DIFFERING_GRIDS);
470 }
471
472 const size_t newStartIndex = std::max(_startIndex, s._startIndex);
473 const size_t newEndIndex = std::min(_endIndex, s._endIndex);
474
475 if (newStartIndex >= newEndIndex) {
476 // No overlap, return empty Support.
477 return createEmpty(_grid);
478 } else {
479 return Support(_grid, newStartIndex, newEndIndex);
480 }
