51 template <typename T, size_t sizea, size_t sizeb>
52 static T evaluateInterval(const std::array<T, sizea> &a,
53 const std::array<T, sizeb> &b, const T &dxhalf) {
54 std::array<T, (sizea + sizeb) / 2> coefficients;
55 coefficients.fill(static_cast<T>(0));
56 for (size_t i = 0; i < sizea; i++) {
57 for (size_t j = i % 2; j < sizeb; j += 2) {
58 coefficients[(i + j) / 2] += a[i] * b[j];
59 }
60 }
61
62 // Use Horner's scheme to evaluate.
63 constexpr int endIndex = static_cast<int>(coefficients.size()) - 1;
64 static_assert(endIndex >= 0);
65
66 T result = coefficients[endIndex] / static_cast<T>(2 * endIndex + 1);
67 const T dxhalf_squared = dxhalf * dxhalf;
68
69 for (int i = endIndex - 1; i >= 0; i--) {
70 result =
71 result * dxhalf_squared + coefficients[i] / static_cast<T>(2 * i + 1);
72 }
73 return static_cast<T>(2) * dxhalf * result;
74 }

108 template <typename T, size_t ordera, size_t orderb>
109 T evaluate(const Spline<T, ordera> &a, const Spline<T, orderb> &b) const {
110 // Will also check whether the two grids are equivalent.
111 const support::Support integrandSupport =
112 a.getSupport().calcIntersection(b.getSupport());
113 const size_t nintervals = integrandSupport.numberOfIntervals();
114
115 const auto &grid = integrandSupport.getGrid();
116
117 T result = static_cast<T>(0);
118
119 for (size_t interv = 0; interv < nintervals; interv++) {
120 const auto absIndex = integrandSupport.absoluteFromRelative(interv);
121
122 const auto aIndex =
123 a.getSupport().intervalIndexFromAbsolute(absIndex).value();
124 const auto bIndex =
125 b.getSupport().intervalIndexFromAbsolute(absIndex).value();
126
127 const T dxhalf = (a.getSupport()[aIndex + 1] - a.getSupport()[aIndex]) /
128 static_cast<T>(2);
129
130 result += evaluateInterval(
131 _o1.transform(a.getCoefficients()[aIndex], grid, absIndex),
132 _o2.transform(b.getCoefficients()[bIndex], grid, absIndex), dxhalf);
133 }
134 return result;
135 }
